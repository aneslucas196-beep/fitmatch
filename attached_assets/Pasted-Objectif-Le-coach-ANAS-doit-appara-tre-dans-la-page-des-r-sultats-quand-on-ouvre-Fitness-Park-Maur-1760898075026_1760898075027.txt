// === Helpers de normalisation & debug ===
function normalize(s) {
  return (s || '')
    .toString()
    .normalize('NFD')
    .replace(/\p{Diacritic}/gu, '')
    .toLowerCase()
    .trim();
}
function logBlock(title, payload) {
  console.log(`\n==== ${title} ====`);
  try { console.log(JSON.stringify(payload, null, 2)); } catch { console.log(payload); }
}

// === Après avoir chargé `gyms` et `coaches` ===
const gymsById = Object.fromEntries(gyms.map(g => [g.id, g]));
const gymIdByName = Object.fromEntries(gyms.map(g => [normalize(g.name), g.id]));
const gymIdByCP   = Object.fromEntries(gyms.map(g => [normalize(g.postal_code), g.id]));

// Sécuriser / corriger coach.gyms (accepte ID ou nom de salle)
// Et forcer la cohérence publique/active par défaut
coaches.forEach(c => {
  if (!Array.isArray(c.gyms)) c.gyms = [];
  c.gyms = c.gyms.map(g => {
    const n = normalize(g);
    if (gymsById[g]) return g;                     // déjà un ID
    if (gymIdByName[n]) return gymIdByName[n];     // nom → ID
    if (gymIdByCP[n])   return gymIdByCP[n];       // CP  → ID
    return g; // fallback (sera filtré si inconnu)
  });
  if (c.public === undefined) c.public = true;
  if (!c.status) c.status = 'active';
});

Remplace/installe la fonction qui retourne les coachs d’une salle par ID :

export function getCoachesForGym(gymId, opts = {}) {
  const { specialty = null } = opts;
  const res = coaches
    .filter(c => c && c.public && c.status === 'active')
    .filter(c => Array.isArray(c.gyms) && c.gyms.some(id => normalize(id) === normalize(gymId)))
    .filter(c => !specialty || normalize(c.specialty) === normalize(specialty));

  logBlock('RESULT_FOR_GYM', {
    gymId,
    specialty,
    count: res.length,
    coach_ids: res.map(c => c.id)
  });
  return res;
}
