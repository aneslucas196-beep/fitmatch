  <script>
    // Accordéon minimal façon Planity (flèche qui pivote + panneau qui se déplie)
    const items = document.querySelectorAll('.item');

    items.forEach(item => {
      const btn = item.querySelector('.btn');
      const panel = item.querySelector('.panel');

      btn.addEventListener('click', () => {
        const isOpen = item.classList.contains('open');

        // Ferme les autres
        items.forEach(i => {
          i.classList.remove('open');
          const p = i.querySelector('.panel');
          const b = i.querySelector('.btn');
          p.style.maxHeight = 0;
          p.hidden = true;
          b.setAttribute('aria-expanded', 'false');
        });

        // Ouvre celui cliqué
        if (!isOpen) {
          item.classList.add('open');
          panel.hidden = false;
          panel.style.maxHeight = panel.scrollHeight + 'px';
          btn.setAttribute('aria-expanded', 'true');
        }
      });
    });

    // Support ouverture initiale via hash (optionnel)
    if (location.hash) {
      const target = document.querySelector(location.hash);
      if (target && target.classList.contains('item')) {
        target.querySelector('.btn').click();
        target.scrollIntoView({behavior:'smooth', block:'start'});
      }
    }

    // Recalcule la hauteur au redimensionnement
    window.addEventListener('resize', () => {
      document.querySelectorAll('.item.open .panel').forEach(p => {
        p.style.maxHeight = p.scrollHeight + 'px';
      });
    });
  </script>
