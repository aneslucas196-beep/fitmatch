Corrige la fonction qui récupère les coachs d’une salle (remplace/ajuste ta fonction existante) :

function getCoachesForGym(gymId, opts = {}) {
  const { specialty = null } = opts;

  const result = coaches
    .filter(c => c && (c.public !== false))               // public true par défaut
    .filter(c => (c.status || 'active') === 'active')     // status par défaut "active"
    .filter(c => Array.isArray(c.gyms) && c.gyms.some(g => normalize(g) === normalize(gymId)))
    .filter(c => !specialty || normalize(c.specialty) === normalize(specialty));

  debugLog('RESULT_FOR_GYM', { gymId, specialty, count: result.length, ids: result.map(c => c.id) });
  return result;
}

const gymsByName = Object.fromEntries(gyms.map(g => [normalize(g.name), g.id]));
const allGymIds = new Set(gyms.map(g => g.id));

coaches.forEach(c => {
  if (!Array.isArray(c.gyms)) c.gyms = [];
  c.gyms = c.gyms.map(g => {
    const ng = normalize(g);
    // si déjà un id connu
    if (allGymIds.has(g)) return g;
    // si c'est un nom, map vers l'id
    if (gymsByName[ng]) return gymsByName[ng];
    return g; // fallback
  });
});
