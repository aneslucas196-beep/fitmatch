  <script>
    let map, placesService, autocomplete, center, markers = [];
    let nextPageToken = null;
    let currentRequest = null;

    function kmDist(a, b) {
      const R = 6371, dLat = (b.lat - a.lat) * Math.PI/180, dLng = (b.lng - a.lng) * Math.PI/180;
      const s1 = Math.sin(dLat/2), s2 = Math.sin(dLng/2);
      const A = s1*s1 + Math.cos(a.lat*Math.PI/180)*Math.cos(b.lat*Math.PI/180)*s2*s2;
      return R * 2 * Math.atan2(Math.sqrt(A), Math.sqrt(1-A));
    }

    function clearMarkers() { markers.forEach(m => m.setMap(null)); markers = []; }

    function renderResults(list) {
      const container = document.getElementById('results');
      container.innerHTML = "";
      list.forEach(p => {
        const d = kmDist(center, { lat: p.geometry.location.lat(), lng: p.geometry.location.lng() });
        const card = document.createElement('div'); card.className = 'card';
        card.innerHTML = `
          <div class="title">${p.name}</div>
          <div class="meta">
            ${p.vicinity || p.formatted_address || ''}<br/>
            ${p.rating ? `⭐ ${p.rating} (${p.user_ratings_total || 0}) · ` : ''}${d.toFixed(1)} km
          </div>`;
        container.appendChild(card);
      });
    }

    function addMarkers(list) {
      clearMarkers();
      list.forEach(p => {
        const m = new google.maps.Marker({ map, position: p.geometry.location, title: p.name });
        markers.push(m);
      });
    }

    function doNearbySearch(append=false) {
      const radius = parseInt(document.getElementById('radius').value, 10);
      const req = {
        location: new google.maps.LatLng(center.lat, center.lng),
        radius,
        type: ['gym'],
        keyword: 'fitness musculation crossfit'
      };
      currentRequest = req;

      placesService.nearbySearch(req, (results, status, pagination) => {
        if (status !== google.maps.places.PlacesServiceStatus.OK || !results) {
          document.getElementById('results').innerHTML = "<div class='meta'>Aucune salle trouvée.</div>";
          document.getElementById('more').style.display = 'none';
          return;
        }
        // Trier par distance
        results.sort((a,b) => {
          const da = kmDist(center, { lat:a.geometry.location.lat(), lng:a.geometry.location.lng() });
          const db = kmDist(center, { lat:b.geometry.location.lat(), lng:b.geometry.location.lng() });
          return da - db;
        });

        if (!append) { renderResults(results); addMarkers(results); }
        else {
          // append
          const container = document.getElementById('results');
          results.forEach(p => {
            const d = kmDist(center, { lat: p.geometry.location.lat(), lng: p.geometry.location.lng() });
            const card = document.createElement('div'); card.className = 'card';
            card.innerHTML = `<div class="title">${p.name}</div>
              <div class="meta">${p.vicinity || p.formatted_address || ''}<br/>${p.rating ? `⭐ ${p.rating} (${p.user_ratings_total || 0}) · ` : ''}${d.toFixed(1)} km</div>`;
            container.appendChild(card);
          });
          // ne pas re-placer les marqueurs pour les pages suivantes (simple)
        }

        nextPageToken = pagination && pagination.hasNextPage ? pagination.nextPage() && pagination : null;
        // Hack Google : il faut attendre ~2s pour que pagination.nextPage fonctionne
        document.getElementById('more').style.display = nextPageToken ? 'block' : 'none';
      });
    }

    function initMap() {
      const defaultCenter = { lat: 48.8566, lng: 2.3522 }; // Paris
      center = defaultCenter;

      map = new google.maps.Map(document.getElementById('map'), { center, zoom: 12 });
      placesService = new google.maps.places.PlacesService(map);

      // Autocomplete sur l'input adresse
      const countrySel = document.getElementById('country');
      const addr = document.getElementById('address');
      const langSel = document.getElementById('lang');

      const acOptions = { fields: ['geometry', 'formatted_address'] };
      autocomplete = new google.maps.places.Autocomplete(addr, acOptions);
      autocomplete.addListener('place_changed', () => {
        const place = autocomplete.getPlace();
        if (place?.geometry?.location) {
          center = { lat: place.geometry.location.lat(), lng: place.geometry.location.lng() };
          map.setCenter(center); map.setZoom(13);
        }
      });

      // Restreindre l’Autocomplétion au pays choisi
      countrySel.addEventListener('change', () => {
        const v = countrySel.value;
        autocomplete.setComponentRestrictions(v ? { country: [v] } : {});
      });

      // Bouton Chercher
      const goBtn = document.getElementById('go');
      goBtn.addEventListener('click', async () => {
        goBtn.disabled = true;

        // Si l'utilisateur n'a pas choisi via Autocomplete, géocoder l'adresse entrée
        if (addr.value && (!center || center === defaultCenter)) {
          try {
            const url = `https://maps.googleapis.com/maps/api/geocode/json?address=${encodeURIComponent(addr.value)}${countrySel.value ? `&components=country:${countrySel.value}` : ''}&key=${GOOGLE_API_KEY}`;
            const r = await fetch(url);
            const g = await r.json();
            if (g.status === 'OK' && g.results && g.results[0]) {
              const loc = g.results[0].geometry.location;
              center = { lat: loc.lat, lng: loc.lng };
              map.setCenter(center); map.setZoom(13);
            }
          } catch(e) { console.warn('Geocode fallback error', e); }
        }

        doNearbySearch(false);
        // délai pour permettre la pagination de Google
        setTimeout(() => { goBtn.disabled = false; }, 1500);
      });

      // Bouton Voir plus
      document.getElementById('more').addEventListener('click', () => {
        // pagination.nextPage est déjà “bindée” par Google; on rappelle doNearbySearch via le callback interne
        // Petite astuce : relancer une recherche append après ~2s
        setTimeout(() => doNearbySearch(true), 2000);
      });
    }

    // Astuce: injecter la clé depuis un <script> avant celui de Maps
    const GOOGLE_API_KEY = "YOUR_API_KEY"; // Replit peut l’injecter si besoin
    window.initMap = initMap;
  </script>
