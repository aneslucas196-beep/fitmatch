<script>
/* === CONFIG === */
const COACH_ID = "c_demo";         // ➜ remplace par l’ID réel
const GYM_ID   = "g_demo";          // ➜ optionnel
const SLOT_MINUTES = 30;
const DAY_START = "10:00";          // 10h -> modifiable
const DAY_END   = "19:00";          // 19h -> modifiable

/* === HELPERS === */
const $ = s => document.querySelector(s);
const pad = n => String(n).padStart(2,"0");
const fmtDate = d => `${d.toLocaleDateString('fr-FR',{weekday:'short'})} ${pad(d.getDate())} ${d.toLocaleDateString('fr-FR',{month:'short'})}`;
const toISO = (d) => d.toISOString();
function addMinutes(date, m){ const d=new Date(date); d.setMinutes(d.getMinutes()+m); return d; }
function setTime(base, hhmm){const [h,m]=hhmm.split(':').map(Number);const d=new Date(base);d.setHours(h,m,0,0);return d;}
function between(date, start, end){return date>=start && date<end;}

/* === MOCK API (remplace ces 3 fonctions par tes vraies routes) === */
async function fetchAvailability(coachId, fromISO, toISO){
  // renvoie des intervalles disponibles (UTC ISO)
  const from=new Date(fromISO), to=new Date(toISO);
  const items=[];
  for(let d=new Date(from); d<=to; d.setDate(d.getDate()+1)){
    // tous les jours mêmes tranches (10:00-13:00, 14:00-18:00)
    const s1=setTime(d, "10:00"), e1=setTime(d,"13:00");
    const s2=setTime(d, "14:00"), e2=setTime(d,"18:00");
    items.push({start:s1.toISOString(), end:e1.toISOString()});
    items.push({start:s2.toISOString(), end:e2.toISOString()});
  }
  return items;
}
async function fetchBookings(coachId, fromISO, toISO){
  // créneaux déjà pris (exemple)
  return [
    // un créneau pris demain à 11:00-11:30
    (()=>{const d=new Date(); d.setDate(d.getDate()+1);
      const s=setTime(d,"11:00"); const e=addMinutes(s,SLOT_MINUTES);
      return {start:s.toISOString(), end:e.toISOString()}
    })()
  ];
}
async function createBooking(payload){
  // appelle ton POST /api/bookings ici
  console.log("BOOKING ▶", payload);
  await new Promise(r=>setTimeout(r,400)); // petite latence
  return {ok:true,id:"bk_"+Math.random().toString(36).slice(2)};
}

/* === STATE === */
let weekOffset = 0;
let selection = null; // {start, end, dayIndex}

/* === RENDER === */
async function renderWeek(){
  const today = new Date(); today.setHours(0,0,0,0);
  const start = new Date(today); start.setDate(start.getDate() + (weekOffset*7));
  const end = new Date(start); end.setDate(end.getDate()+6);

  // header dates
  $("#rangeText").textContent =
    `${start.toLocaleDateString('fr-FR')} – ${end.toLocaleDateString('fr-FR')}`;

  const headRow = $("#daysHead");
  headRow.innerHTML = "<th></th>";
  const days = [];
  for(let i=0;i<7;i++){
    const d = new Date(start); d.setDate(d.getDate()+i);
    days.push(d);
    const th = document.createElement("th");
    th.innerHTML = `<div>${fmtDate(d)}<small>${d.getFullYear()}</small></div>`;
    headRow.appendChild(th);
  }

  // slots axis
  const times = [];
  let cursor = setTime(start, DAY_START);
  const dayEnd = setTime(start, DAY_END);
  while(cursor < dayEnd){
    times.push(cursor.toTimeString().slice(0,5));
    cursor = addMinutes(cursor, SLOT_MINUTES);
  }

  // fetch data
  const avail = await fetchAvailability(COACH_ID, start.toISOString(), end.toISOString());
  const bookings = await fetchBookings(COACH_ID, start.toISOString(), end.toISOString());
  const body = $("#gridBody"); body.innerHTML = "";

  // function to check status for given date+time
  function statusFor(dtStart){
    const dtEnd = addMinutes(dtStart, SLOT_MINUTES);
    // past
    if(dtStart < new Date()) return "disabled";
    // inside availability?
    const inAvail = avail.some(a=>{
      const s=new Date(a.start), e=new Date(a.end);
      return between(dtStart,s,e) && dtEnd<=e;
    });
    if(!inAvail) return "disabled";
    // busy?
    const isBusy = bookings.some(b=>{
      const s=new Date(b.start), e=new Date(b.end);
      return !(dtEnd<=s || dtStart>=e); // overlap
    });
    return isBusy ? "busy" : "free";
  }

  // build rows
  times.forEach(time=>{
    const tr = document.createElement("tr");
    const th = document.createElement("td");
    th.textContent = time;
    tr.appendChild(th);

    days.forEach((d,dayIndex)=>{
      const td = document.createElement("td");
      const slotStart = setTime(d, time);
      const st = statusFor(slotStart);
      const btn = document.createElement("div");
      btn.className = `slot ${st}`;
      btn.textContent = time;

      if(st==="free"){
        btn.addEventListener("click", ()=>{
          // clear previous
          document.querySelectorAll(".slot.selected").forEach(el=>el.classList.remove("selected"));
          btn.classList.add("selected");
          selection = {
            start: slotStart,
            end: addMinutes(slotStart,SLOT_MINUTES),
            dayIndex
          };
          $("#recap").textContent =
            `Séance le ${d.toLocaleDateString('fr-FR',{weekday:'long',day:'2-digit',month:'long'})} à ${time}`;
          $("#confirmBtn").disabled = false;
        });
      }

      td.appendChild(btn);
      tr.appendChild(td);
    });

    body.appendChild(tr);
  });

  // reset if week changed
  selection = null;
  $("#recap").textContent = "Aucun créneau sélectionné";
  $("#confirmBtn").disabled = true;
}

$("#prevWeek").addEventListener("click", ()=>{ weekOffset--; renderWeek(); });
$("#nextWeek").addEventListener("click", ()=>{ weekOffset++; renderWeek(); });

$("#confirmBtn").addEventListener("click", async ()=>{
  if(!selection) return;
  const payload = {
    coachId: COACH_ID,
    gymId: GYM_ID,
    startsAt: toISO(selection.start),
    endsAt: toISO(selection.end),
    source: "fitmatch"
  };
  const res = await createBooking(payload);
  if(res?.ok){
    showToast("Réservation confirmée ✅");
    // recharger la semaine pour marquer le créneau en “occupé”
    renderWeek();
  }else{
    showToast("Erreur de réservation ❌");
  }
});

function showToast(msg){
  const t=$("#toast"); t.textContent=msg; t.classList.add("show");
  setTimeout(()=>t.classList.remove("show"), 2200);
}

/* boot */
renderWeek();
