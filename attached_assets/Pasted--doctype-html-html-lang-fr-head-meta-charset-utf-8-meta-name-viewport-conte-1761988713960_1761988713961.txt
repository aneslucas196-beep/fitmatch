  <script>
    // -------------------------------
    // Mini utilitaires d’auth
    // -------------------------------
    const KEY = 'fm_user';

    function getUser(){
      try { return JSON.parse(localStorage.getItem(KEY)); }
      catch(e){ return null; }
    }
    function setUser(user){
      localStorage.setItem(KEY, JSON.stringify(user));
    }
    function logout(){
      localStorage.removeItem(KEY);
      location.reload();
    }
    function go(path){
      // mémorise l’URL courante pour le retour post-auth
      sessionStorage.setItem('redirect_to', location.pathname + location.search + location.hash);
      location.href = path;
    }

    // -------------------------------
    // Rendu identification
    // -------------------------------
    function renderIdentification(){
      const user = getUser();
      const idLogged = document.getElementById('id-logged');
      const idGuest  = document.getElementById('id-guest');
      const idGuestActions = document.getElementById('id-guest-actions');
      const idDivider = document.getElementById('id-divider');

      if(user && user.email){
        // Affiche Nom + Email, cache les boutons
        document.getElementById('id-name').textContent = user.name || 'Compte FitMatch';
        document.getElementById('id-email').textContent = user.email;

        idLogged.classList.remove('hide');
        idGuest.classList.add('hide');
        idGuestActions.classList.add('hide');
        idDivider.classList.add('hide');
      } else {
        // Version non connectée
        idLogged.classList.add('hide');
        idGuest.classList.remove('hide');
        idGuestActions.classList.remove('hide');
        idDivider.classList.remove('hide');
      }
    }

    // -------------------------------
    // Actions
    // -------------------------------
    document.getElementById('btn-signup').addEventListener('click', ()=> go('/signup.html'));
    document.getElementById('btn-login').addEventListener('click',  ()=> go('/login.html'));
    document.getElementById('btn-logout').addEventListener('click', logout);
    document.getElementById('btn-edit').addEventListener('click', ()=> go('/account.html'));
    document.getElementById('btn-confirm').addEventListener('click', ()=>{
      alert('✅ Réservation confirmée (démo).');
    });

    // Init
    renderIdentification();
  </script>
