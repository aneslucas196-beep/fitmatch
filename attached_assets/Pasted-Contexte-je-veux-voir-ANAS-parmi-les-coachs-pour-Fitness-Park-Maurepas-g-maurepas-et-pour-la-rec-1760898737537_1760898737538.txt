2) Normalisation & mapping au chargement

Après le chargement des JSON (gyms, coaches), ajouter :

function norm(s){return (s||'').toString().normalize('NFD').replace(/\p{Diacritic}/gu,'').toLowerCase().trim();}
const gymsById   = Object.fromEntries(gyms.map(g=>[g.id,g]));
const idByName   = Object.fromEntries(gyms.map(g=>[norm(g.name),g.id]));
const idByCP     = Object.fromEntries(gyms.map(g=>[norm(g.postal_code),g.id]));

// sécuriser & remapper coach.gyms (accepte id, nom salle, CP)
coaches.forEach(c=>{
  if(!Array.isArray(c.gyms)) c.gyms=[];
  c.gyms=c.gyms.map(g=>{
    if(gymsById[g]) return g;                 // déjà un id
    const n=norm(g);
    if(idByName[n]) return idByName[n];       // nom → id
    if(idByCP[n])   return idByCP[n];         // CP  → id
    return g;
  });
  if(c.public===undefined) c.public=true;
  if(!c.status) c.status='active';
});

3) Fonction de recherche (désactive filtres piégeux)

Remplacer la fonction qui récupère les coachs d’une salle par :

export function getCoachesForGym(gymId,{specialty=null}={}){
  const list=coaches
    .filter(c=>c && c.status==='active' && c.public!==false)
    .filter(c=>Array.isArray(c.gyms) && c.gyms.some(id=>id===gymId))
    .filter(c=>!specialty || norm(c.specialty)===norm(specialty));

  // HOTFIX: si ANAS appartient à cette salle mais absent (tri/pagination), on l’injecte
  const anas=coaches.find(c=>c.id==='c_anas');
  const anasBelongs=anas && Array.isArray(anas.gyms) && anas.gyms.includes(gymId);
  const missing=anasBelongs && !list.some(c=>c.id==='c_anas');
  if(missing) list.push(anas);

  // désactiver toute pagination stricte ici (laisser le composant UI paginer après)
  return list; // pas de slice() ici
}
