Objectif : Après l’inscription (nom complet + e-mail + mot de passe), afficher un écran “Confirmation par e-mail” (pas de téléphone).
Exigences :


Générer et envoyer un code à 6 chiffres sur l’e-mail saisi (endpoint POST /api/send-otp-email avec { email }).


Afficher une interface propre (carte blanche) : titre, message, champ “Code de vérification”, lien “Recevoir le code à nouveau”, lien “Modifier l’e-mail”, bouton “Enregistrer”, bouton “Se déconnecter”.


Vérifier le code via POST /api/verify-otp avec { email, code }.


Si l’API n’est pas encore prête, fallback local : générer le code côté front, le stocker dans localStorage et le “simuler” (console + petite bannière “Code envoyé à …”).


À la vérification OK : marquer l’utilisateur vérifié (ex. localStorage.fitmatch.user = { fullName, email, verified:true }), rester sur la même page de réservation et remplacer la section Identification par le bloc identité (Nom + Email + “Modifier / Se déconnecter”).


Bouton “Modifier” rouvre le formulaire d’inscription.


Design cohérent avec nos cartes FitMatch (bordures douces, fond blanc, centres, responsive).

<script>
/**
 * Vérification par e-mail — FitMatch
 * - Utilise localStorage si l’API n’est pas prête (mode démo)
 * - S’intègre avec fitmatch.user (déjà créé lors de l’inscription)
 */

const LS_USER_KEY = 'fitmatch.user';
const LS_OTP_KEY  = 'fitmatch.otp'; // { email, code, expiresAt }

const overlay   = document.getElementById('email-verify-overlay');
const evEmail   = document.getElementById('evEmailHint');
const evCode    = document.getElementById('evCode');
const evSubmit  = document.getElementById('evSubmit');
const evResend  = document.getElementById('evResend');
const evEdit    = document.getElementById('evEditEmail');
const evLogout  = document.getElementById('evLogout');
const evToast   = document.getElementById('evToast');

/* —— Helpers UI —— */
function showOverlay(){ overlay.classList.remove('hidden'); }
function hideOverlay(){ overlay.classList.add('hidden'); }
function toast(msg){ evToast.textContent = msg; evToast.classList.remove('hidden'); setTimeout(()=>evToast.classList.add('hidden'), 3500); }

/* —— OTP côté front (fallback) —— */
function generateCode(){ return ('' + Math.floor(100000 + Math.random()*900000)); }
function saveOtp(email, code){
  const expiresAt = Date.now() + 5*60*1000; // 5 minutes
  localStorage.setItem(LS_OTP_KEY, JSON.stringify({ email, code, expiresAt }));
}
function getOtp(){ try{ return JSON.parse(localStorage.getItem(LS_OTP_KEY)||'null'); }catch{ return null; } }
function clearOtp(){ localStorage.removeItem(LS_OTP_KEY); }

/* —— Envoi OTP —— */
async function sendOtpEmail(email){
  // === OPTION A : API réelle ===
  // const res = await fetch('/api/send-otp-email', {
  //   method:'POST', headers:{'Content-Type':'application/json'},
  //   body: JSON.stringify({ email })
  // });
  // if(!res.ok) throw new Error('Envoi OTP impossible');

  // === OPTION B : Fallback local (démo) ===
  const code = generateCode();
  saveOtp(email, code);
  console.log('%c[DEV] Code OTP envoyé à ' + email + ' : ' + code, 'color: #16a34a; font-weight:700;');
  toast('Code envoyé à ' + email + ' (vérifie ta boîte mail).');
}

/* —— Vérification OTP —— */
async function verifyOtpEmail(email, code){
  // === OPTION A : API réelle ===
  // const res = await fetch('/api/verify-otp', {
  //   method:'POST', headers:{'Content-Type':'application/json'},
  //   body: JSON.stringify({ email, code })
  // });
  // if(!res.ok) throw new Error('Code invalide');

  // === OPTION B : Fallback local ===
  const otp = getOtp();
  if(!otp || otp.email !== email) throw new Error("Aucun code actif pour cet e-mail.");
  if(Date.now() > otp.expiresAt) throw new Error("Le code a expiré. Renvoyez-le.");
  if(otp.code !== code) throw new Error("Code incorrect.");
  // ok
}

/* —— Flux principal —— */
function openEmailVerification(){
  const u = JSON.parse(localStorage.getItem(LS_USER_KEY) || 'null');
  if(!u || !u.email){ return; }
  evEmail.textContent = u.email;
  evCode.value = '';
  showOverlay();
}

evResend.addEventListener('click', async ()=>{
  const u = JSON.parse(localStorage.getItem(LS_USER_KEY) || 'null');
  if(!u) return;
  try{
    await sendOtpEmail(u.email);
  }catch(e){ toast('Impossible d’envoyer le code. Réessaie.'); }
});

evEdit.addEventListener('click', ()=>{
  // On ferme l’overlay et on ré-affiche le formulaire d’inscription
  hideOverlay();
  const form = document.getElementById('signupForm');
  const summary = document.getElementById('id-summary');
  if(form && summary){ summary.classList.add('hidden'); form.classList.remove('hidden'); }
  toast('Tu peux modifier ton e-mail puis recréer ton compte.');
});

evLogout.addEventListener('click', ()=>{
  localStorage.removeItem(LS_USER_KEY);
  clearOtp();
  hideOverlay();
  // Ré-afficher le formulaire si présent
  const form = document.getElementById('signupForm');
  const summary = document.getElementById('id-summary');
  if(form && summary){ summary.classList.remove('hidden'); summary.classList.add('hidden'); }
  location.reload();
});

evSubmit.addEventListener('click', async ()=>{
  const u = JSON.parse(localStorage.getItem(LS_USER_KEY) || 'null');
  if(!u) return;
  const code = evCode.value.trim();
  if(code.length !== 6){ toast('Entre le code à 6 chiffres.'); return; }

  evSubmit.disabled = true; evSubmit.textContent = 'Vérification…';
  try{
    await verifyOtpEmail(u.email, code);
    // MAJ utilisateur → vérifié
    localStorage.setItem(LS_USER_KEY, JSON.stringify({ ...u, verified:true }));
    clearOtp();
    hideOverlay();

    // Rafraîchir l’affichage Identification si présent
    if(typeof renderIdentification === 'function'){ renderIdentification(); }
    toast('Adresse e-mail vérifiée ✅');
  }catch(e){
    toast(e.message || 'Code invalide.');
  }finally{
    evSubmit.disabled = false; evSubmit.textContent = 'Enregistrer';
  }
});

/* —— Hooke l’inscription existante : envoyer OTP après “Créer mon compte” —— */
/* Si tu utilises le code d’inscription que je t’ai donné précédemment,
   ajoute ce listener APRÈS avoir stocké fitmatch.user */
(function attachAfterSignup(){
  const originalSignup = window.signupViaAPI;
  // Décore la fonction si elle existe
  if(typeof originalSignup === 'function'){
    window.signupViaAPI = async function(payload){
      const res = await originalSignup(payload); // crée le compte (API ou local)
      try{
        const u = JSON.parse(localStorage.getItem(LS_USER_KEY) || 'null');
        if(u && u.email){
          await sendOtpEmail(u.email);
          openEmailVerification();
        }
      }catch{}
      return res;
    };
  }else{
    // Si pas de fonction globale, on ouvre quand même après création locale
    document.addEventListener('fitmatch:signup:done', async ()=>{
      const u = JSON.parse(localStorage.getItem(LS_USER_KEY) || 'null');
      if(u && u.email){
        await sendOtpEmail(u.email);
        openEmailVerification();
      }
    });
  }
})();
</script>


Comment ça va se passer pour toi


L’utilisateur remplit Nom + E-mail + Mot de passe puis clique Créer mon compte.


Un code à 6 chiffres est “envoyé” sur son e-mail (via ton API, ou en démo dans la console).


L’écran “Confirmation par e-mail” s’ouvre : il tape le code → Enregistrer.


Si OK : l’utilisateur passe en vérifié, on reste sur la même page et la section “Identification” affiche son Nom + E-mail (comme tu voulais).


Il peut renvoyer le code, modifier l’e-mail, ou se déconnecter.


Quand tu auras ton backend, remplace les parties “OPTION B : Fallback local” par tes appels réels send-otp-email et verify-otp.
